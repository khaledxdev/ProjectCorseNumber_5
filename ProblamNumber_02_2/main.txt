#include<iostream>
#include<cmath>
using namespace std;

enum enPrimNotPrim{ Prim = 1 , NotPrim = 2 };

int ReadPositiveNumber(string Message)
{
	int Number = 0; 
	do {
		cout << Message << endl;
		cin >> Number;
	} while (Number <= 0);
	
	return Number;

}

enPrimNotPrim CheckPrim(int Number)
{
	if (Number == 1) return enPrimNotPrim::NotPrim;

	int M  =   round (Number / 2);



	for (int i = 2; i <= M; i++) {
		if(Number % i  == 0 )
		{
			return enPrimNotPrim::NotPrim;
		}
	}

	return enPrimNotPrim::Prim;

}

void PrintPrim()
{
	int Number = ReadPositiveNumber("Please Enter Your a Positive Number ");
	enPrimNotPrim Result = CheckPrim(Number);
	
	


	if ( Result == enPrimNotPrim::Prim) {

		cout <<Number<< " is Prim Number" << endl;
	}
	else {
		cout << Number << " is Not Prim Number" << endl;

	}


}

int main() 
{
	PrintPrim();


}
